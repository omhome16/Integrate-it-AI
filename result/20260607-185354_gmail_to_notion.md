```python
import os
import logging
import httpx
from tenacity import retry, wait_exponential, stop_after_attempt
from typing import Dict, List
import argparse
import json

# Set up logging
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

# Load environment variables
GMAIL_API_KEY = os.environ.get('GMAIL_API_KEY')
NOTION_API_KEY = os.environ.get('NOTION_API_KEY')
if not GMAIL_API_KEY or not NOTION_API_KEY:
    raise ValueError("GMAIL_API_KEY and NOTION_API_KEY environment variables are required")

# Define field mappings
FIELD_MAPPINGS = [
    {'sourceField': 'id', 'sourceType': 'string', 'targetField': 'id', 'targetType': 'string', 'transformation': 'direct', 'confidence': 0.9},
    {'sourceField': 'snippet', 'sourceType': 'string', 'targetField': 'title', 'targetType': 'string', 'transformation': 'direct', 'confidence': 0.8},
    {'sourceField': 'historyId', 'sourceType': 'string', 'targetField': 'created_time', 'targetType': 'string', 'transformation': 'direct', 'confidence': 0.7},
    {'sourceField': 'labelIds', 'sourceType': 'array', 'targetField': 'tags', 'targetType': 'array', 'transformation': 'direct', 'confidence': 0.6},
    # Omitting 'payload.headers' mapping as it's semantically unsafe to directly map email headers to Notion properties
    {'sourceField': 'payload.body.text', 'sourceType': 'string', 'targetField': 'content', 'targetType': 'string', 'transformation': 'direct', 'confidence': 0.8},
    {'sourceField': 'payload.from', 'sourceType': 'string', 'targetField': 'author', 'targetType': 'string', 'transformation': 'direct', 'confidence': 0.9},
    {'sourceField': 'payload.to', 'sourceType': 'string', 'targetField': 'assignee', 'targetType': 'string', 'transformation': 'direct', 'confidence': 0.8},
    # Omitting 'payload.cc' and 'payload.bcc' mappings as they're not directly applicable to Notion pages
    {'sourceField': 'payload.subject', 'sourceType': 'string', 'targetField': 'subject', 'targetType': 'string', 'transformation': 'direct', 'confidence': 0.9},
    {'sourceField': 'internalDate', 'sourceType': 'string', 'targetField': 'last_edited_time', 'targetType': 'string', 'transformation': 'direct', 'confidence': 0.8},
]

# Define API clients
async def get_gmail_threads(client: httpx.AsyncClient, page_token: str = None) -> Dict:
    params = {'pageToken': page_token} if page_token else {}
    response = await client.get('https://www.googleapis.com/gmail/v1/users/me/threads', params=params, headers={'Authorization': f'Bearer {GMAIL_API_KEY}'})
    response.raise_for_status()
    return response.json()

async def get_gmail_thread(client: httpx.AsyncClient, thread_id: str) -> Dict:
    response = await client.get(f'https://www.googleapis.com/gmail/v1/users/me/threads/{thread_id}', headers={'Authorization': f'Bearer {GMAIL_API_KEY}'})
    response.raise_for_status()
    return response.json()

async def get_notion_pages(client: httpx.AsyncClient, start_cursor: str = None) -> Dict:
    params = {'start_cursor': start_cursor} if start_cursor else {}
    response = await client.get('https://api.notion.com/v1/pages', params=params, headers={'Authorization': f'Bearer {NOTION_API_KEY}', 'Content-Type': 'application/json'})
    response.raise_for_status()
    return response.json()

async def create_notion_page(client: httpx.AsyncClient, page: Dict) -> Dict:
    response = await client.post('https://api.notion.com/v1/pages', headers={'Authorization': f'Bearer {NOTION_API_KEY}', 'Content-Type': 'application/json'}, json=page)
    response.raise_for_status()
    return response.json()

async def update_notion_page(client: httpx.AsyncClient, page_id: str, page: Dict) -> Dict:
    response = await client.patch(f'https://api.notion.com/v1/pages/{page_id}', headers={'Authorization': f'Bearer {NOTION_API_KEY}', 'Content-Type': 'application/json'}, json=page)
    response.raise_for_status()
    return response.json()

# Define field mapping functions
def map_fields(gmail_thread: Dict) -> Dict:
    notion_page = {}
    for mapping in FIELD_MAPPINGS:
        if mapping['sourceField'] in gmail_thread:
            notion_page[mapping['targetField']] = gmail_thread[mapping['sourceField']]
    return notion_page

# Define sync function
@retry(wait=wait_exponential(multiplier=1, min=4, max=10), stop=stop_after_attempt(3))
async def sync_gmail_to_notion(dry_run: bool = False) -> None:
    async with httpx.AsyncClient() as client:
        ok_count = 0
        failed_count = 0
        # Get Gmail threads
        threads_response = await get_gmail_threads(client)
        while True:
            for thread in threads_response.get('threads', []):
                try:
                    # Get Gmail thread
                    gmail_thread = await get_gmail_thread(client, thread['id'])
                    # Map fields
                    notion_page = map_fields(gmail_thread)
                    # Check if page exists in Notion
                    pages_response = await get_notion_pages(client)
                    existing_page = next((page for page in pages_response.get('results', []) if page.get('id') == notion_page.get('id')), None)
                    if existing_page:
                        # Update existing page
                        if not dry_run:
                            await update_notion_page(client, existing_page['id'], notion_page)
                        logger.info(f'Updated page {existing_page["id"]}')
                        ok_count += 1
                    else:
                        # Create new page
                        if not dry_run:
                            await create_notion_page(client, notion_page)
                        logger.info(f'Created page {notion_page.get("id")}')
                        ok_count += 1
                except Exception as e:
                    logger.error(f'Error syncing thread {thread["id"]}: {str(e)}')
                    failed_count += 1
            if 'nextPageToken' not in threads_response:
                break
            threads_response = await get_gmail_threads(client, threads_response['nextPageToken'])
        logger.info(f'Sync complete: {ok_count} successful, {failed_count} failed')

# Define CLI
if __name__ == '__main__':
    parser = argparse.ArgumentParser(description='Sync Gmail to Notion')
    parser.add_argument('--dry-run', action='store_true', help='Dry run mode')
    args = parser.parse_args()
    import asyncio
    asyncio.run(sync_gmail_to_notion(args.dry_run))
```
